46template<typename _Scalar, std::size_t _Rows, std::size_t _Cols>
47struct StaticMatrix : public std::array<std::array<_Scalar, _Cols>, _Rows>
48{
50 static constexpr std::size_t Rows = _Rows;
52 static constexpr std::size_t Cols = _Cols;
53
54 using std::array<std::array<_Scalar, _Cols>, _Rows>::operator[];
55
59 constexpr StaticMatrix() : std::array<std::array<_Scalar, _Cols>, _Rows>{}
60 {
61 for (auto i = 0u; i != _Rows; ++i) { operator[](i).fill(_Scalar(0)); }
62 }

67 template<std::size_t _NRows, std::size_t _NCols>
68 constexpr StaticMatrix<_Scalar, _NRows, _NCols> block(std::size_t row0, std::size_t col0) const
69 {
70 StaticMatrix<_Scalar, _NRows, _NCols> ret;
71 for (auto i = 0u; i < _NRows; ++i) {
72 for (auto j = 0u; j < _NCols; ++j) { ret[i][j] = operator[](row0 + i)[col0 + j]; }
73 }
74 return ret;
75 }

80 constexpr StaticMatrix<_Scalar, _Rows, _Cols>
81 operator+(const StaticMatrix<_Scalar, _Rows, _Cols> & o) const
82 {
83 StaticMatrix<_Scalar, _Rows, _Cols> ret;
84 for (auto i = 0u; i < _Rows; ++i) {
85 for (auto j = 0u; j < _Cols; ++j) { ret[i][j] = operator[](i)[j] + o[i][j]; }
86 }
87 return ret;
88 }

93 constexpr StaticMatrix<_Scalar, _Rows, _Cols> transpose() const
94 {
95 StaticMatrix<_Scalar, _Rows, _Cols> ret;
96 for (auto i = 0u; i < _Rows; ++i) {
97 for (auto j = 0u; j < _Cols; ++j) { ret[j][i] = operator[](i)[j]; }
98 }
99 return ret;
100 }
101
105 template<std::size_t _NCols>
106 constexpr StaticMatrix<_Scalar, _Rows, _NCols>
107 operator*(const StaticMatrix<_Scalar, _Cols, _NCols> & o) const
108 {
109 StaticMatrix<_Scalar, _Rows, _NCols> ret;
110 for (auto i = 0u; i < _Rows; ++i) {
111 for (auto j = 0u; j < _NCols; ++j) {
112 for (auto k = 0u; k < _Cols; ++k) { ret[i][j] += operator[](i)[k] * o[k][j]; }
113 }
114 }
115 return ret;
116 }
