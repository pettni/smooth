46 struct CeresParameterizationFunctor
47 {
48  template<typename Scalar>
49  bool operator()(const Scalar * x, const Scalar * delta, Scalar * x_plus_delta) const
50  {
51  using GScalar = decltype(G().template cast<Scalar>());
52 
53  Eigen::Map<const GScalar> mx(x);
54  Eigen::Map<const typename GScalar::Tangent> mdelta(delta);
55  Eigen::Map<GScalar> mx_plus_delta(x_plus_delta);
56 
57  mx_plus_delta = mx + mdelta;
58  return true;
59  }
60 };

79 template<typename _F, typename _Wrt>
80 auto dr_ceres(_F && f, _Wrt && x)
81 {
82  // There is potential to improve thie speed of this by reducing casting.
83  // The ceres Jet type supports binary operations with e.g. double, but currently
84  // the Lie operations require everything to have a uniform scalar type. Enabling
85  // plus and minus for different scalars would thus save some casts.
86  using Result = typename decltype(std::apply(f, x))::PlainObject;
87  using Scalar = typename Result::Scalar;
88 
89  const Result fval = std::apply(f, x);
90 
91  static constexpr Eigen::Index Nx = utils::tuple_dof<_Wrt>::value;
92  const auto nx = std::apply([](auto &&... args) { return (args.size() + ...); }, x);
93  static constexpr Eigen::Index Ny = Result::SizeAtCompileTime;
94  const auto ny = fval.size();
95 
96  static_assert(Nx != -1, "Ceres autodiff only supports static sizes");
97 
98  Eigen::Matrix<Scalar, Nx, 1> a(nx);
99  a.setZero();
100 
101  Eigen::Matrix<Scalar, Ny, 1> b(ny);
102 
103  Eigen::Matrix<Scalar, Ny, Nx, (Nx == 1) ? Eigen::ColMajor : Eigen::RowMajor> jac(ny, nx);
104 
105  const Scalar * a_ptr[1] = {a.data()};
106  Scalar * jac_ptr[1] = {jac.data()};
107 
108  const auto f_deriv = [&]<typename T>(const T * in, T * out) {
109  Eigen::Map<const Eigen::Matrix<T, Nx, 1>> mi(in, nx);
110  Eigen::Map<Eigen::Matrix<T, Ny, 1>> mo(out, ny);
111  mo = std::apply(f, utils::tuple_plus(utils::tuple_cast<T>(x), mi)) - fval.template cast<T>();
112  return true;
113  };
114 
115  ceres::internal::AutoDifferentiate<Result::SizeAtCompileTime,
116  ceres::internal::StaticParameterDims<Nx>>(f_deriv, a_ptr, b.size(), b.data(), jac_ptr);
117 
118  return std::make_pair(std::move(fval), Eigen::Matrix<Scalar, Ny, Nx>(jac));
119 }
