64auto dr_ceres(auto && f, auto && x)
65{
66 // There is potential to improve thie speed of this by reducing casting.
67 // The ceres Jet type supports binary operations with e.g. double, but currently
68 // the Lie operations require everything to have a uniform scalar type. Enabling
69 // plus and minus for different scalars would thus save some casts.
70 using Result = decltype(std::apply(f, x));
71 using Scalar = ::smooth::Scalar<Result>;
72
73 static_assert(Manifold<Result>, "f(x) is not a Manifold");
74
75 Result fval = std::apply(f, x);
76
77 static constexpr Eigen::Index Nx = wrt_Dof<decltype(x)>();
78 static constexpr Eigen::Index Ny = Dof<Result>;
79 const Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x);
80 const Eigen::Index ny = dof<Result>(fval);
81
82 static_assert(Nx > 0, "Ceres autodiff does not support dynamic sizes");
83
84 Eigen::Matrix<Scalar, Nx, 1> a = Eigen::Matrix<Scalar, Nx, 1>::Zero(nx);
85 Eigen::Matrix<Scalar, Ny, 1> b(ny);
86 Eigen::Matrix<Scalar, Ny, Nx, (Nx == 1) ? Eigen::ColMajor : Eigen::RowMajor> jac(ny, nx);
87
88 const auto f_deriv = [&]<typename T>(const T * in, T * out) {
89 Eigen::Map<const Eigen::Matrix<T, Nx, 1>> mi(in, nx);
90 Eigen::Map<Eigen::Matrix<T, Ny, 1>> mo(out, ny);
91 mo = rminus<CastT<T, Result>>(std::apply(f, wrt_rplus(wrt_cast<T>(x), mi)), cast<T, Result>(fval));
92 return true;
93 };
94 const Scalar * a_ptr[1] = {a.data()};
95 Scalar * jac_ptr[1] = {jac.data()};
96
97 ceres::internal::AutoDifferentiate<Dof<Result>, ceres::internal::StaticParameterDims<Nx>>(
98 f_deriv, a_ptr, static_cast<int>(b.size()), b.data(), jac_ptr);
99
100 return std::make_pair(std::move(fval), Eigen::Matrix<Scalar, Ny, Nx>(jac));
101}
