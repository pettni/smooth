52 template<typename M>
53 concept Manifold =
54 requires
55 {
56  typename M::Scalar;
57  typename M::PlainObject;
58  {M::SizeAtCompileTime}->std::convertible_to<Eigen::Index>; // degrees of freedom at compile time
59 } &&
60 requires(const M & m1, const M & m2, const Eigen::Matrix<typename M::Scalar, M::SizeAtCompileTime, 1> & a)
61 {
62  {m1.size()}->std::convertible_to<Eigen::Index>; // degrees of freedom at runtime
63  {m1 + a}->std::convertible_to<typename M::PlainObject>;
64  {m1 - m2}->std::convertible_to<Eigen::Matrix<typename M::Scalar, M::SizeAtCompileTime, 1>>;
65  {m1.template cast<double>()};
66 };

73 template<typename G>
74 concept LieGroup = Manifold<G> &&
75 // static constants
76 requires {
77  typename G::Tangent;
78  {G::Dof}->std::convertible_to<Eigen::Index>;
79 } &&
80 (G::Dof >= 1) &&
81 (G::SizeAtCompileTime == G::Dof) &&
82 (G::Tangent::SizeAtCompileTime == G::Dof) &&
83 // member methods
84 requires(const G & g1, const G & g2)
85 {
86  {g1.inverse()}->std::convertible_to<typename G::PlainObject>;
87  {g1 * g2}->std::convertible_to<typename G::PlainObject>;
88  {g1.log()}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, 1>>;
89  {g1.Ad()}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
90 } &&
91 // static methods
92 requires(const Eigen::Matrix<typename G::Scalar, G::Dof, 1> & a)
93 {
94  {G::Identity()}->std::convertible_to<typename G::PlainObject>;
95  {G::exp(a)}->std::convertible_to<typename G::PlainObject>;
96  {G::ad(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
97 };
