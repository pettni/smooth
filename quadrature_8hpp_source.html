90template<std::size_t K, std::size_t I = 8>
91 requires(K <= 40)
92constexpr std::pair<std::array<double, K>, std::array<double, K>> lgr_nodes()
93{
94 // initial guess
95 auto xs = cgr_nodes<K>();
96
97 // initialize weights
98 std::array<double, K> ws;
99 ws[0] = 2. / (K * K);
100
101 // two rightmost column of legendre basis matrix
102 constexpr auto B =
103 polynomial_basis<PolynomialBasis::Legendre, K>().template block<K + 1, 2>(0, K - 1);
104
105 for (auto i = 1u; i < K; ++i) {
106 StaticMatrix<double, 1, 2> U_B;
107 for (auto iter = 0u; iter < I; ++iter) {
108 U_B = monomial_derivative<K>(xs[i], 0) * B;
109 const auto dU_B = monomial_derivative<K>(xs[i], 1) * B;
110 const double fxi = U_B[0][0] + U_B[0][1];
111 const double dfxi = dU_B[0][0] + dU_B[0][1];
112 xs[i] -= fxi / dfxi;
113 }
114 ws[i] = (1. - xs[i]) / (K * K * U_B[0][0] * U_B[0][0]);
115 }
116
117 return {xs, ws};
118}
