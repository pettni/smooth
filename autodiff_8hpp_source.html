54 template<typename _F, typename _Wrt>
55 auto dr_autodiff(_F && f, _Wrt && x)
56 {
57  using Result = decltype(std::apply(f, x));
58  using Scalar = typename Result::Scalar;
59  using AdScalar = autodiff::forward::Dual<Scalar, Scalar>;
60 
61  // determine sizes if input and output
62  constexpr auto Nx = utils::tuple_dof<_Wrt>::value;
63  auto nx = std::apply([](auto &&... args) { return (args.size() + ...); }, x);
64  static constexpr auto Ny = Result::SizeAtCompileTime;
65 
66  auto val = std::apply(f, x);
67 
68  // cast val and x to ad types
69  auto x_ad = utils::tuple_cast<AdScalar>(x);
70  typename decltype(val.template cast<AdScalar>())::PlainObject val_ad =
71  val.template cast<AdScalar>();
72 
73  // zero-valued tangent element
74  Eigen::Matrix<AdScalar, Nx, 1> a_ad = Eigen::Matrix<AdScalar, Nx, 1>::Zero(nx);
75 
76  Eigen::Matrix<Scalar, Ny, Nx> jac = autodiff::forward::jacobian(
77  [&f, &val_ad, &x_ad](Eigen::Matrix<AdScalar, Nx, 1> & var) -> Eigen::Matrix<AdScalar, Ny, 1> {
78  return std::apply(f, utils::tuple_plus(x_ad, var)) - val_ad;
79  },
80  wrt(a_ad),
81  wrt(a_ad));
82 
83  return std::make_pair(val, jac);
84 }
