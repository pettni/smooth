148  G eval(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
149  {
150  // index of relevant interval
151  int64_t istar = static_cast<int64_t>((t - t0_) / dt_);
152 
153  double u;
154  // clamp to end of range if necessary
155  if (istar < 0) {
156  istar = 0;
157  u = 0;
158  } else if (istar + K + 1 > ctrl_pts_.size()) {
159  istar = ctrl_pts_.size() - K - 1;
160  u = 1;
161  } else {
162  u = (t - t0_ - istar * dt_) / dt_;
163  }
164 
165  constexpr auto Mstatic = detail::cum_coefmat<CSplineType::BSPLINE, double, K>().transpose();
166  Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> M(Mstatic[0].data());
167 
168  G g = cspline_eval<K, G>(
169  ctrl_pts_ | std::views::drop(istar) | std::views::take(K + 1), M, u, vel, acc);
170 
171  if (vel.has_value()) { vel.value() /= dt_; }
172  if (acc.has_value()) { acc.value() /= (dt_ * dt_); }
173 
174  return g;
175  }

196 template<std::size_t K, std::ranges::range Rt, std::ranges::range Rg>
197 BSpline<K, std::ranges::range_value_t<Rg>> fit_bspline(const Rt & tt, const Rg & gg, double dt)
198 {
199  static_assert(LieGroup<std::ranges::range_value_t<Rg>>, "Rg value type is LieGroup");
200  static_assert(std::is_same_v<std::ranges::range_value_t<Rt>, double>, "Rt value type is double");
201 
202  using G = std::ranges::range_value_t<Rg>;
203 
204  auto [tmin_ptr, tmax_ptr] = std::minmax_element(std::ranges::begin(tt), std::ranges::end(tt));
205 
206  const double t0 = *tmin_ptr;
207  const double t1 = *tmax_ptr;
208 
209  const std::size_t NumData = std::min(std::ranges::size(tt), std::ranges::size(gg));
210  const std::size_t NumPts = K + static_cast<std::size_t>((t1 - t0 + dt) / dt);
211 
212  constexpr auto Mstatic = detail::cum_coefmat<CSplineType::BSPLINE, double, K>().transpose();
213  Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> M(Mstatic[0].data());
214 
215  auto f = [&](const auto & var) {
216  Eigen::VectorXd ret(G::Dof * NumData);
217 
218  Eigen::SparseMatrix<double, Eigen::RowMajor> Jac;
219  Jac.resize(G::Dof * NumData, G::Dof * NumPts);
220  Jac.reserve(Eigen::Matrix<int, -1, 1>::Constant(G::Dof * NumData, G::Dof * (K + 1)));
221 
222  auto t_iter = std::ranges::begin(tt);
223  auto g_iter = std::ranges::begin(gg);
224 
225  for (auto i = 0u; i != NumData; ++t_iter, ++g_iter, ++i) {
226  const int64_t istar = static_cast<int64_t>((*t_iter - t0) / dt);
227  const double u = (*t_iter - t0 - istar * dt) / dt;
228 
229  Eigen::Matrix<double, G::Dof, (K + 1) * G::Dof> d_vali_pts;
230  auto g_spline = cspline_eval<K, G>(
231  var | std::views::drop(istar) | std::views::take(K + 1), M, u, {}, {}, d_vali_pts);
232 
233  const typename G::Tangent resi = g_spline - *g_iter;
234 
235  ret.segment<G::Dof>(i * G::Dof) = resi;
236 
237  const Eigen::Matrix<double, G::Dof, G::Dof> d_resi_vali = G::dr_expinv(resi);
238  const Eigen::Matrix<double, G::Dof, (K + 1) * G::Dof> d_resi_pts = d_resi_vali * d_vali_pts;
239 
240  for (auto r = 0u; r != G::Dof; ++r) {
241  for (auto c = 0u; c != G::Dof * (K + 1); ++c) {
242  Jac.insert(i * G::Dof + r, istar * G::Dof + c) = d_resi_pts(r, c);
243  }
244  }
245  }
246 
247  Jac.makeCompressed();
248 
249  return std::make_pair(std::move(ret), Eigen::MatrixXd(Jac));
250  };
251 
252  // create optimization variable
253  ManifoldVector<G, Eigen::aligned_allocator> ctrl_pts(NumPts);
254 
255  // create initial guess
256  auto t_iter = std::ranges::begin(tt);
257  auto g_iter = std::ranges::begin(gg);
258  for (auto i = 0u; i != NumPts; ++i) {
259  const double t_target = t0 + (i - static_cast<double>(K - 1) / 2) * dt;
260  while (t_iter + 1 < std::ranges::end(tt)
261  && std::abs(t_target - *(t_iter + 1)) < std::abs(t_target - *t_iter)) {
262  ++t_iter;
263  ++g_iter;
264  }
265  ctrl_pts[i] = *g_iter;
266  }
267 
268  // fit to data with loose convergence criteria
269  MinimizeOptions opts;
270  opts.ftol = 1e-3;
271  opts.ptol = 1e-3;
272  opts.max_iter = 10;
273  minimize<diff::Type::ANALYTIC>(f, smooth::wrt(ctrl_pts), opts);
274 
275  return BSpline<K, G>(t0, dt, std::move(ctrl_pts));
276 }
