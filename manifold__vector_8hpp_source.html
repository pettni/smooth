111  template<typename Derived>
112  ManifoldVector<M> & operator+=(const Eigen::MatrixBase<Derived> & a)
113  {

115  for (auto i = 0u; i != this->size(); ++i) {

142  Eigen::VectorX<Scalar<M>> operator-(const ManifoldVector<M> & o) const
143  {
144  std::size_t dof_cnts = 0;
145  if (Dof < M >> 0) {
146  dof_cnts = Dof<M> * size();
147  } else {
148  for (auto i = 0u; i != size(); ++i) { dof_cnts += ::smooth::dof<M>(this->operator[](i)); }
149  }
150 
151  Eigen::VectorX<Scalar<M>> ret(dof_cnts);
152  Eigen::Index idx = 0;
153  for (auto i = 0u; i != size(); ++i) {
154  const auto & size_i = ::smooth::dof<M>(this->operator[](i));
155  ret.template segment<Dof<M>>(idx, size_i) = rminus<M>(this->operator[](i), o[i]);
156  idx += size_i;
157  }
