165 {
166 Eigen::Matrix<Scalar, 4, 10> ret = Eigen::Matrix<Scalar, 4, 10>::Zero();
167
168 // v transformation
169 ret.template block<3, 3>(0, 0) = v(3) * so3().matrix();
170 ret.template block<3, 3>(0, 3) = so3().matrix();
171 ret.template block<3, 1>(0, 6) = r3_v();
172 ret.template block<3, 3>(0, 7) = so3().dr_action(v.template segment<3>(0));
