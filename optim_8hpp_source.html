58template<diff::Type D>
59void minimize(auto && f, auto && x, const MinimizeOptions & opts = MinimizeOptions{})
60{
61 // evaluate residuals and jacobian at initial point
62 auto [r, J] = diff::dr<1, D>(f, x);
63
64 using JType = std::decay_t<decltype(J)>;
65
66 // extract some properties from jacobian
67 static constexpr bool is_sparse = std::is_base_of_v<Eigen::SparseMatrixBase<JType>, JType>;
68 static constexpr auto Nx = JType::ColsAtCompileTime;
69 const auto nx = J.cols();
70
71 // scaling parameters
72 Eigen::Vector<double, Nx> d(nx);
73 if constexpr (is_sparse) {
74 d = (Eigen::RowVector<double, JType::RowsAtCompileTime>::Ones(J.rows()) * J.cwiseProduct(J))
75 .cwiseSqrt()
76 .transpose();
77 } else {
78 d = J.colwise().stableNorm().transpose();
79 }
80
81 // ensure scaling parameters are non-zero
82 for (auto i = 0u; i != d.size(); ++i) {
83 if (d[i] == 0) { d[i] = 1; }
84 }
85
86 double r_norm = r.stableNorm();
87 double Delta = 100. * d.stableNorm(); // TODO for Rn arguments we should multiply with norm(x)
88
89 std::size_t iter = 0u;
90
91 auto t0 = std::chrono::high_resolution_clock::now();
92
93 if (opts.verbose) {
94 using std::cout, std::left, std::setw, std::right;
95 // clang-format off
96 cout << "================= Levenberg-Marquardt Solver ================\n";
97 cout << "Solving NLSQ with n=" << J.cols() << ", m=" << J.rows() << '\n';
98 cout << setw(8) << right << "ITER"
99 << setw(14) << right << "|| r ||"
100 << setw(14) << right << "RED"
101 << setw(14) << right << "|| D * a ||"
102 << setw(10) << right << "TIME" << '\n';
103 // clang-format on
104 }
105
106 for (; iter < opts.max_iter; ++iter) {
107 // calculate step a via LM parameter algorithm
108 Eigen::Vector<double, Nx> a(nx);
109 double lambda;
110 if constexpr (is_sparse) {
111 std::tie(lambda, a) = detail::lmpar_sparse(J, d, r, Delta);
112 } else {
113 std::tie(lambda, a) = detail::lmpar(J, d, r, Delta);
114 }
115
116 // evaluate function and jacobian at x + a
117 const auto x_plus_a = wrt_rplus(x, a);
118 const auto [r_cand, J_cand] = diff::dr<1, D>(f, x_plus_a);
119
120 const double r_cand_norm = r_cand.stableNorm();
121 const double Da_norm = d.cwiseProduct(a).stableNorm();
122
123 // calculate actual to predicted reduction
124 const Eigen::Vector<double, JType::RowsAtCompileTime> Ja = J * a;
125 const double act_red = 1. - Eigen::numext::abs2(r_cand_norm / r_norm);
126 const double fra2 = Eigen::numext::abs2(Ja.stableNorm() / r_norm);
127 const double fra3 = Eigen::numext::abs2(std::sqrt(lambda) * Da_norm / r_norm);
128 const double pred_red = fra2 + 2. * fra3;
129 const double rho = act_red / pred_red;
130
131 // update trust region following Mor√© (1978)
132 if (rho < 0.25) {
133 double mu = 0.5;
134 if (r_cand_norm <= r_norm) {
135 // leave at 0.5
136 } else if (r_cand_norm <= 10 * r_norm) {
137 const double gamma = -fra2 - fra3;
138 mu = std::clamp(gamma / (2. * gamma + act_red), 0.1, 0.5);
139 } else {
140 mu = 0.1;
141 }
142 Delta *= mu;
143 } else if ((lambda == 0 && rho < 0.75) || rho > 0.75) {
144 Delta = 2 * Da_norm;
145 }
146
147 if (opts.verbose) {
148 using std::cout, std::setw, std::right;
149 // clang-format off
150 cout << setw(7) << right << iter << ":"
151 << std::scientific
152 << setw(14) << right << r_cand_norm
153 << setw(14) << right << act_red
154 << setw(14) << right << Da_norm
155 << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
156 << '\n';
157 // clang-format on
158 }
159
161
162 if (rho > 1e-4) {
163 x = x_plus_a;
164 r = r_cand;
165 J = J_cand;
166 r_norm = r_cand_norm;
167
168 // update scaling
169 if constexpr (is_sparse) {
170 d = (Eigen::RowVector<double, JType::RowsAtCompileTime>::Ones(J.rows()) * J.cwiseProduct(J))
171 .cwiseSqrt()
172 .transpose();
173 } else {
174 d = d.cwiseMax(J.colwise().stableNorm().transpose());
175 }
176 }
177
179
180 // function tolerance
181 if (std::abs(act_red) < opts.ftol && pred_red < opts.ftol && rho <= 2.) { break; }
182
183 // parameter tolerance
184 // TODO a.size() should be norm(x) for non-angle states
185 if (Da_norm < opts.ptol * static_cast<double>(a.size())) { break; }
186 }
187
189 if (opts.verbose) {
190 using std::cout, std::left, std::right, std::setw;
191
192 // clang-format off
193 cout << "NLSQ solver summary:\n";
194 cout << setw(25) << left << "Iterations" << setw(10) << right << iter << '\n';
195 cout << setw(25) << left << "Total time (microseconds)" << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count() << '\n';
196 cout << "=============================================================\n";
197 // clang-format on
198 }
199}
