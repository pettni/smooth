100  const double fra2 = Eigen::numext::abs2((J * a).stableNorm() / r_norm);
101  const double fra3 = Eigen::numext::abs2(std::sqrt(lambda) * Da_norm / r_norm);
102  const double pred_red = fra2 + 2. * fra3;
103  const double rho = act_red / pred_red;

109  mu = 0.5;
110  } else if (r_cand_norm <= 10 * r_norm) {

113  } else {

115  }
116  Delta *= mu;
117  } else if ((lambda == 0 && rho < 0.75) || rho > 0.75) {
118  Delta = 2 * Da_norm;
119  }

132  .cwiseSqrt()
133  .transpose());

136  }
