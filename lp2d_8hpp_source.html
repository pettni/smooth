54template<std::ranges::range R>
55inline std::tuple<Scalar, Scalar, Status>
56solve(Scalar cx, Scalar cy, const R & rows) requires(std::tuple_size_v<std::ranges::range_value_t<R>> == 3)
57{
58 const Scalar sqnorm = cx * cx + cy * cy;
59
60 if (sqnorm < detail::eps) { return {0, 0, Status::Optimal}; }
61
62 if (std::ranges::empty(rows)) { return {0, 0, Status::DualInfeasible}; }
63
64 // transformation matrix:
65 // [x; y] = [cP -sP; sP cP] [xt; yt]
66 const Scalar cP = cy / sqnorm;
67 const Scalar sP = -cx / sqnorm;
68
69 // insert rotated halfplanes with unit vector norm 1
70 std::vector<detail::HalfPlane> input;
71 input.reserve(std::ranges::size(rows));
72 for (const auto [a, b, c] : rows) {
73 const double ra = cP * a + sP * b;
74 const double rb = -sP * a + cP * b;
75
76 const double norm = ra * ra + rb * rb;
77
78 if (norm > detail::eps && c < detail::inf) {
79 input.push_back(detail::HalfPlane{
80 .a = ra / norm,
81 .b = rb / norm,
82 .c = c / norm,
83 .active = true,
84 });
85 }
86 }
87
88 // scale factor
89 Scalar lambda{1};
90 for (const auto & hp : input) { lambda = std::max(lambda, std::abs(hp.c)); }
91
92 for (auto & hp : input) { hp.c /= lambda; }
93
94 const auto [xt_opt, yt_opt, status] = solve_impl(input);
95
96 // multiplication that returns 0 for 0 * inf (regular multiplication returns nan)
97 const auto mul = [](Scalar a, Scalar b) { return std::abs(a) > detail::eps ? a * b : 0; };
98
99 // return solution in original coordinates
100 return {
101 lambda * (mul(cP, xt_opt) - mul(sP, yt_opt)),
102 lambda * (mul(sP, xt_opt) + mul(cP, yt_opt)),
103 status,
104 };
105}

113// value and derivative
114using ValDer = std::tuple<Scalar, Scalar>;

116// value and subderivative (upper,lower)
117using ValSubDer = std::tuple<Scalar, Scalar, Scalar>;

119// halfplanes that define upper and lower bounds on y (as a function of x)
120inline constexpr auto active_y_upper = [](const auto & hp) { return hp.active && hp.b > 0; };
121inline constexpr auto active_y_lower = [](const auto & hp) { return hp.active && hp.b < 0; };
122
123// halfplane as bounds on y
124inline constexpr auto hp_to_yslope = [](const HalfPlane & hp, Scalar x) -> std::pair<Scalar, Scalar> {
125 // ax + by <=> c for b != 0 <==> y <=> c/b - (a/b) x
126 const Scalar alpha = -hp.a / hp.b;
127 const Scalar beta = hp.c / hp.b;
128
129 // ensure we can evaluate at \pm inf
130 if (std::abs(alpha) <= eps) { return {beta, alpha}; }
131
132 return {alpha * x + beta, alpha};
133};

138inline std::optional<Scalar> intersection(const HalfPlane & hp1, const HalfPlane & hp2)
139{
140 const Scalar lhs = hp1.a * hp2.b - hp2.a * hp1.b;
141 if (std::abs(lhs) > eps) { return (hp2.b * hp1.c - hp1.b * hp2.c) / lhs; }
142 return {};
143}

145// g(x) = max { ai * x + bi }, and its subderivative
146inline const auto gfun = [](const std::ranges::range auto & hps, const Scalar x) -> ValSubDer {
147 ValSubDer ret{-inf, 0, 0};
148 for (const auto & hp : hps | std::views::filter(active_y_lower)) {
149 const auto [yx, dyx] = hp_to_yslope(hp, x);
150 if (std::get<0>(ret) > yx + eps) {
151 // do noting
152 } else if (yx > std::get<0>(ret) + eps) {
153 ret = {yx, dyx, dyx};
154 } else {
155 std::get<1>(ret) = std::min(std::get<1>(ret), dyx);
156 std::get<2>(ret) = std::max(std::get<2>(ret), dyx);
157 }
158 }
159 return ret;
160};
161
162// h(x) = min { a * x + b }, and its subderivative
163inline const auto hfun = [](const std::ranges::range auto & hps, const Scalar x) -> ValSubDer {
164 ValSubDer ret{inf, 0, 0};
165 for (const auto & hp : hps | std::views::filter(active_y_upper)) {
166 const auto [yx, dyx] = hp_to_yslope(hp, x);
167 if (std::get<0>(ret) + eps < yx) {
168 // do noting
169 } else if (yx + eps < std::get<0>(ret)) {
170 ret = {yx, dyx, dyx};
171 } else {
172 std::get<1>(ret) = std::min(std::get<1>(ret), dyx);
173 std::get<2>(ret) = std::max(std::get<2>(ret), dyx);
174 }
175 }
176 return ret;
177};

180inline std::optional<Scalar> find_candidate(std::vector<HalfPlane> & hps, Scalar a, Scalar b)
181{
182 std::optional<typename std::vector<HalfPlane>::iterator> it1_store{};
183
184 // keep track of median using two priority queues
185 std::priority_queue<Scalar> small, large;
186 const auto addnum = [&small, &large](Scalar d) {
187 small.push(d);
188 large.push(-small.top());
189 small.pop();
190 while (small.size() < large.size()) {
191 small.push(-large.top());
192 large.pop();
193 }
194 };
195
196 // INTERSECT LOWERS AMONGST THEMSELVES
197
198 for (auto it2 = hps.begin(); it2 != hps.end(); ++it2) {
199
200 if (!active_y_lower(*it2)) { continue; }
201
202 if (!it1_store.has_value()) {
203 it1_store = it2;
204 continue;
205 }
206
207 auto it1 = *it1_store;
208
209 const auto isec = intersection(*it1, *it2);
210
211 int8_t redundant = 0; // 0 (none), 1, or 2
212
213 if (isec.has_value()) {
214 if (a + eps < *isec && *isec + eps < b) {
215 addnum(*isec);
216 it1_store = {};
217 } else { // intersection outside--one is redundant
218 if (a + eps >= *isec) { // check for redundancy at a
219 const auto [v1, dv1] = hp_to_yslope(*it1, a);
220 const auto [v2, dv2] = hp_to_yslope(*it2, a);
221 if (v1 + eps < v2) {
222 redundant = 1;
223 } else if (v2 + eps < v1) {
224 redundant = 2;
225 } else {
226 redundant = dv1 <= dv2 ? 1 : 2;
227 }
228 } else if (*isec + eps >= b) { // check for redundancy at b
229 const auto [v1, dv1] = hp_to_yslope(*it1, b);
230 const auto [v2, dv2] = hp_to_yslope(*it2, b);
231 if (v1 + eps < v2) {
232 redundant = 1;
233 } else if (v2 + eps < v1) {
234 redundant = 2;
235 } else {
236 redundant = dv1 >= dv2 ? 1 : 2;
237 }
238 }
239 }
240 } else { // parallel--so one is redundant
241 redundant = hp_to_yslope(*it1, 0) < hp_to_yslope(*it2, 0) ? 1 : 2;
242 }
243
244 if (redundant == 1) {
245 it1->active = false;
246 it1_store = it2;
247 } else if (redundant == 2) {
248 it2->active = false;
249 }
250 }
251
252 // INTERSECT UPPERS AMONGST THEMSELVES
253
254 it1_store = {};
255
256 for (auto it2 = hps.begin(); it2 != hps.end(); ++it2) {
257
258 if (!active_y_upper(*it2)) { continue; }
259
260 if (!it1_store.has_value()) {
261 it1_store = it2;
262 continue;
263 }
264
265 auto it1 = *it1_store;
266
267 const auto isec = intersection(*it1, *it2);
268
269 int8_t redundant = 0; // 0 (none), 1, or 2
270
271 if (isec.has_value()) {
272 if (a + eps < *isec && *isec + eps < b) {
273 addnum(*isec);
274 it1_store = {};
275 } else { // intersection outside--one is redundant
276 if (a + eps >= *isec) { // check for redundancy at a
277 const auto [v1, dv1] = hp_to_yslope(*it1, a);
278 const auto [v2, dv2] = hp_to_yslope(*it2, a);
279 if (v1 + eps < v2) {
280 redundant = 2;
281 } else if (v2 + eps < v1) {
282 redundant = 1;
283 } else {
284 redundant = dv1 <= dv2 ? 2 : 1;
285 }
286 } else if (*isec + eps >= b) { // check for redundancy at b
287 const auto [v1, dv1] = hp_to_yslope(*it1, b);
288 const auto [v2, dv2] = hp_to_yslope(*it2, b);
289 if (v1 + eps < v2) {
290 redundant = 2;
291 } else if (v2 + eps < v1) {
292 redundant = 1;
293 } else {
294 redundant = dv1 >= dv2 ? 2 : 1;
295 }
296 }
297 }
298 } else { // parallel--so one is redundant
299 redundant = hp_to_yslope(*it1, 0) < hp_to_yslope(*it2, 0) ? 2 : 1;
300 }
301
302 if (redundant == 1) {
303 it1->active = false;
304 it1_store = it2;
305 } else if (redundant == 2) {
306 it2->active = false;
307 }
308 }
309
310 // IF NO POINTS WERE FOUND AND THERE'S A SINGLE LOWER, INTERSECT IT WITH THE UPPERS
311
312 if (small.empty() && std::count_if(hps.cbegin(), hps.cend(), active_y_lower) == 1) {
313 const auto hp_l = *std::find_if(std::ranges::begin(hps), std::ranges::end(hps), active_y_lower);
314 for (auto & hp_u : hps | std::views::filter(active_y_upper)) {
315 const auto isec = intersection(hp_l, hp_u);
316 if (isec.has_value() && a + eps < *isec && *isec + eps < b) { addnum(*isec); }
317 }
318 }
319
320 if (small.empty()) { return {}; }
321
322 // return median element
323 return small.top();
324}

336inline uint8_t check(const std::vector<HalfPlane> & hps, const Scalar x)
337{
338 const auto [gx, sg, Sg] = gfun(hps, x);
339 const auto [hx, sh, Sh] = hfun(hps, x);
340
341 if (gx <= hx + eps) { // FEASIBLE
342 if (gx + eps < hx) { // there's slack, only g matters
343 if (sg > 0) {
344 return 1;
345 } else if (Sg < 0) {
346 return 2;
347 } else {
348 return 0;
349 }
350 } else { // no slack
351 if (sg > 0 && sg >= Sh) {
352 return 1;
353 } else if (Sg < 0 && Sg < sh) {
354 return 2;
355 } else {
356 return 0;
357 }
358 }
359 } else { // INFEASIBLE
360 if (Sg < sh) {
361 return 2;
362 } else if (sg > Sh) {
363 return 1;
364 } else {
365 return 3;
366 }
367 }
368}

381inline std::tuple<Scalar, Scalar, Status> solve_impl(std::vector<HalfPlane> & hps)
382{
383 // halfplanes that define a lower bound on x (independent of y)
384 auto hps_x_lower =
385 hps | std::views::filter([](const auto & hp) { return hp.active && std::abs(hp.b) < eps && hp.a < 0; });
386
387 // initial lower bound on x
388 Scalar a = std::transform_reduce(
389 std::ranges::begin(hps_x_lower),
390 std::ranges::end(hps_x_lower),
391 -inf,
392 [](const Scalar avar, const Scalar bvar) { return std::max(avar, bvar); },
393 [](const HalfPlane & hp) { return hp.c / hp.a; });
394
395 // halfplanes that define an upper bound on x (independent of y)
396 auto hps_x_upper =
397 hps | std::views::filter([](const auto & hp) { return hp.active && std::abs(hp.b) < eps && hp.a > 0; });
398
399 // initial upper bound on x
400 Scalar b = std::transform_reduce(
401 std::ranges::begin(hps_x_upper),
402 std::ranges::end(hps_x_upper),
403 inf,
404 [](const Scalar avar, const Scalar bvar) { return std::min(avar, bvar); },
405 [](const HalfPlane & hp) { return hp.c / hp.a; });
406
407 // we remove at least one halfplane per iterations, so need at most N iterations
408 for (auto iter = hps.size(); iter > 0; --iter) {
409 const auto x = find_candidate(hps, a, b);
410
411 // remove hps that were marked as not active
412 hps.erase(std::remove_if(hps.begin(), hps.end(), [](const auto & hp) { return !hp.active; }), hps.end());
413
414 if (!x.has_value()) { break; }
415
416 switch (check(hps, *x)) {
417 case 0:
418 return {*x, std::get<0>(gfun(hps, *x)), Status::Optimal};
419 break;
420 case 1:
421 b = *x;
422 break;
423 case 2:
424 a = *x;
425 break;
426 case 3:
427 return {0, inf, Status::PrimaryInfeasible};
428 break;
429 }
430 }
431
432 // no intersection points, only need to consider boundaries
433 const auto [ga, sga, Sga] = gfun(hps, a);
434 const auto [ha, sha, Sha] = hfun(hps, a);
435
436 const auto [gb, sgb, Sgb] = gfun(hps, b);
437 const auto [hb, shb, Shb] = hfun(hps, b);
438
439 if (ga == ha && gb == hb && std::abs(ga) == inf && std::abs(gb) == inf) {
440 // special case where bounds are equal and \pm inf
441 if (gfun(hps, 0) <= hfun(hps, 0)) {
442 return {0., 0., Status::DualInfeasible};
443 } else {
444 return {0., 0., Status::PrimaryInfeasible};
445 }
446 }
447
448 if (ga > ha && gb > hb) {
449 return {0, 0, Status::PrimaryInfeasible};
450 } else if ((ga <= ha && gb > hb) || ga < gb) {
451 return {a, ga, ga == -inf ? Status::DualInfeasible : Status::Optimal};
452 } else {
453 return {b, gb, gb == -inf ? Status::DualInfeasible : Status::Optimal};
454 }
455}
