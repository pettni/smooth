78 template<std::ranges::range R>
79 std::tuple<Scalar, Scalar, Status> solve(Scalar cx, Scalar cy, const R & rows) requires(
80  std::tuple_size_v<std::ranges::range_value_t<R>> == 3)
81 {
82  const Scalar sqnorm = cx * cx + cy * cy;
83 
84  if (sqnorm < detail::eps) { return {0, 0, Status::Optimal}; }
85 
86  if (std::ranges::empty(rows)) { return {0, 0, Status::DualInfeasible}; }
87 
88  // transformation matrix:
89  // [x; y] = [cP -sP; sP cP] [xt; yt]
90  const Scalar cP = cy / sqnorm;
91  const Scalar sP = -cx / sqnorm;
92 
93  // insert rotated halfplanes with unit vector norm 1
94  std::vector<detail::HalfPlane> input;
95  input.reserve(std::ranges::size(rows));
96  for (const auto [a, b, c] : rows) {
97  const double ra = cP * a + sP * b;
98  const double rb = -sP * a + cP * b;
99 
100  const double norm = ra * ra + rb * rb;
101 
102  if (norm > detail::eps && c < detail::inf) {
103  input.push_back(detail::HalfPlane{
104  .a = ra / norm,
105  .b = rb / norm,
106  .c = c / norm,
107  .active = true,
108  });
109  }
110  }
111 
112  // scale factor
113  Scalar lambda{1};
114  for (const auto & hp : input) { lambda = std::max(lambda, std::abs(hp.c)); }
115 
116  for (auto & hp : input) { hp.c /= lambda; }
117 
118  const auto [xt_opt, yt_opt, status] = solve_impl(input);
119 
120  // multiplication that returns 0 for 0 * inf (regular multiplication returns nan)
121  const auto mul = [](Scalar a, Scalar b) { return std::abs(a) > detail::eps ? a * b : 0; };
122 
123  // return solution in original coordinates
124  return {
125  lambda * (mul(cP, xt_opt) - mul(sP, yt_opt)),
126  lambda * (mul(sP, xt_opt) + mul(cP, yt_opt)),
127  status,
128  };
129 }

138  // value and derivative
139  using ValDer = std::tuple<Scalar, Scalar>;
140 
141  // value and subderivative (upper,lower)
142  using ValSubDer = std::tuple<Scalar, Scalar, Scalar>;
143 
144  // halfplanes that define upper and lower bounds on y (as a function of x)
145  constexpr auto active_y_upper = [](const auto & hp) { return hp.active && hp.b > 0; };
146  constexpr auto active_y_lower = [](const auto & hp) { return hp.active && hp.b < 0; };
147 
148  // halfplane as bounds on y
149  constexpr auto hp_to_yslope = [](const HalfPlane & hp, Scalar x) -> std::pair<Scalar, Scalar> {
150  // ax + by <=> c for b != 0 <==> y <=> c/b - (a/b) x
151  const Scalar alpha = -hp.a / hp.b;
152  const Scalar beta = hp.c / hp.b;
153 
154  // ensure we can evaluate at \pm inf
155  if (std::abs(alpha) <= eps) { return {beta, alpha}; }
156 
157  return {alpha * x + beta, alpha};
158  };

163  const std::optional<Scalar> intersection(const HalfPlane & hp1, const HalfPlane & hp2)
164  {
165  const Scalar lhs = hp1.a * hp2.b - hp2.a * hp1.b;
166  if (std::abs(lhs) > eps) { return (hp2.b * hp1.c - hp1.b * hp2.c) / lhs; }
167  return {};
168  }
169 
170  // g(x) = max { ai * x + bi }, and its subderivative
171  const auto gfun = [](const std::ranges::range auto & hps, const Scalar x) -> ValSubDer {
172  ValSubDer ret{-inf, 0, 0};
173  for (const auto & hp : hps | std::views::filter(active_y_lower)) {
174  const auto [yx, dyx] = hp_to_yslope(hp, x);
175  if (std::get<0>(ret) > yx + eps) {
176  // do noting
177  } else if (yx > std::get<0>(ret) + eps) {
178  ret = {yx, dyx, dyx};
179  } else {
180  std::get<1>(ret) = std::min(std::get<1>(ret), dyx);
181  std::get<2>(ret) = std::max(std::get<2>(ret), dyx);
182  }
183  }
184  return ret;
185  };
186 
187  // h(x) = min { a * x + b }, and its subderivative
188  const auto hfun = [](const std::ranges::range auto & hps, const Scalar x) -> ValSubDer {
189  ValSubDer ret{inf, 0, 0};
190  for (const auto & hp : hps | std::views::filter(active_y_upper)) {
191  const auto [yx, dyx] = hp_to_yslope(hp, x);
192  if (std::get<0>(ret) + eps < yx) {
193  // do noting
194  } else if (yx + eps < std::get<0>(ret)) {
195  ret = {yx, dyx, dyx};
196  } else {
197  std::get<1>(ret) = std::min(std::get<1>(ret), dyx);
198  std::get<2>(ret) = std::max(std::get<2>(ret), dyx);
199  }
200  }
201  return ret;
202  };

205  std::optional<Scalar> find_candidate(std::vector<HalfPlane> & hps, Scalar a, Scalar b)
206  {
207  std::optional<typename std::vector<HalfPlane>::iterator> it1_store{};
208 
209  // keep track of median using two priority queues
210  std::priority_queue<Scalar> small, large;
211  const auto addnum = [&small, &large](Scalar d) {
212  small.push(d);
213  large.push(-small.top());
214  small.pop();
215  while (small.size() < large.size()) {
216  small.push(-large.top());
217  large.pop();
218  }
219  };
220 
221  // INTERSECT LOWERS AMONGST THEMSELVES
222 
223  for (auto it2 = hps.begin(); it2 != hps.end(); ++it2) {
224 
225  if (!active_y_lower(*it2)) { continue; }
226 
227  if (!it1_store.has_value()) {
228  it1_store = it2;
229  continue;
230  }
231 
232  auto it1 = *it1_store;
233 
234  const auto isec = intersection(*it1, *it2);
235 
236  int8_t redundant = 0; // 0 (none), 1, or 2
237 
238  if (isec.has_value()) {
239  if (a + eps < *isec && *isec + eps < b) {
240  addnum(*isec);
241  it1_store = {};
242  } else { // intersection outside--one is redundant
243  if (a + eps >= *isec) { // check for redundancy at a
244  const auto [v1, dv1] = hp_to_yslope(*it1, a);
245  const auto [v2, dv2] = hp_to_yslope(*it2, a);
246  if (v1 + eps < v2) {
247  redundant = 1;
248  } else if (v2 + eps < v1) {
249  redundant = 2;
250  } else {
251  redundant = dv1 <= dv2 ? 1 : 2;
252  }
253  } else if (*isec + eps >= b) { // check for redundancy at b
254  const auto [v1, dv1] = hp_to_yslope(*it1, b);
255  const auto [v2, dv2] = hp_to_yslope(*it2, b);
256  if (v1 + eps < v2) {
257  redundant = 1;
258  } else if (v2 + eps < v1) {
259  redundant = 2;
260  } else {
261  redundant = dv1 >= dv2 ? 1 : 2;
262  }
263  }
264  }
265  } else { // parallel--so one is redundant
266  redundant = hp_to_yslope(*it1, 0) < hp_to_yslope(*it2, 0) ? 1 : 2;
267  }
268 
269  if (redundant == 1) {
270  it1->active = false;
271  it1_store = it2;
272  } else if (redundant == 2) {
273  it2->active = false;
274  }
275  }
276 
277  // INTERSECT UPPERS AMONGST THEMSELVES
278 
279  it1_store = {};
280 
281  for (auto it2 = hps.begin(); it2 != hps.end(); ++it2) {
282 
283  if (!active_y_upper(*it2)) { continue; }
284 
285  if (!it1_store.has_value()) {
286  it1_store = it2;
287  continue;
288  }
289 
290  auto it1 = *it1_store;
291 
292  const auto isec = intersection(*it1, *it2);
293 
294  int8_t redundant = 0; // 0 (none), 1, or 2
295 
296  if (isec.has_value()) {
297  if (a + eps < *isec && *isec + eps < b) {
298  addnum(*isec);
299  it1_store = {};
300  } else { // intersection outside--one is redundant
301  if (a + eps >= *isec) { // check for redundancy at a
302  const auto [v1, dv1] = hp_to_yslope(*it1, a);
303  const auto [v2, dv2] = hp_to_yslope(*it2, a);
304  if (v1 + eps < v2) {
305  redundant = 2;
306  } else if (v2 + eps < v1) {
307  redundant = 1;
308  } else {
309  redundant = dv1 <= dv2 ? 2 : 1;
310  }
311  } else if (*isec + eps >= b) { // check for redundancy at b
312  const auto [v1, dv1] = hp_to_yslope(*it1, b);
313  const auto [v2, dv2] = hp_to_yslope(*it2, b);
314  if (v1 + eps < v2) {
315  redundant = 2;
316  } else if (v2 + eps < v1) {
317  redundant = 1;
318  } else {
319  redundant = dv1 >= dv2 ? 2 : 1;
320  }
321  }
322  }
323  } else { // parallel--so one is redundant
324  redundant = hp_to_yslope(*it1, 0) < hp_to_yslope(*it2, 0) ? 2 : 1;
325  }
326 
327  if (redundant == 1) {
328  it1->active = false;
329  it1_store = it2;
330  } else if (redundant == 2) {
331  it2->active = false;
332  }
333  }
334 
335  // IF NO POINTS WERE FOUND AND THERE'S A SINGLE LOWER, INTERSECT IT WITH THE UPPERS
336 
337  if (small.empty() && std::count_if(hps.cbegin(), hps.cend(), active_y_lower) == 1) {
338  const auto hp_l =
339  *std::find_if(std::ranges::begin(hps), std::ranges::end(hps), active_y_lower);
340  for (auto & hp_u : hps | std::views::filter(active_y_upper)) {
341  const auto isec = intersection(hp_l, hp_u);
342  if (isec.has_value() && a + eps < *isec && *isec + eps < b) { addnum(*isec); }
343  }
344  }
345 
346  if (small.empty()) { return {}; }
347 
348  // return median element
349  return small.top();
350  }

362  inline uint8_t check(const std::vector<HalfPlane> & hps, const Scalar x)
363  {
364  const auto [gx, sg, Sg] = gfun(hps, x);
365  const auto [hx, sh, Sh] = hfun(hps, x);
366 
367  if (gx <= hx + eps) { // FEASIBLE
368  if (gx + eps < hx) { // there's slack, only g matters
369  if (sg > 0) {
370  return 1;
371  } else if (Sg < 0) {
372  return 2;
373  } else {
374  return 0;
375  }
376  } else { // no slack
377  if (sg > 0 && sg >= Sh) {
378  return 1;
379  } else if (Sg < 0 && Sg < sh) {
380  return 2;
381  } else {
382  return 0;
383  }
384  }
385  } else { // INFEASIBLE
386  if (Sg < sh) {
387  return 2;
388  } else if (sg > Sh) {
389  return 1;
390  } else {
391  return 3;
392  }
393  }
394  }

407  inline std::tuple<Scalar, Scalar, Status> solve_impl(std::vector<HalfPlane> & hps)
408  {
409  // halfplanes that define a lower bound on x (independent of y)
410  auto hps_x_lower = hps | std::views::filter([](const auto & hp) {
411  return hp.active && std::abs(hp.b) < eps && hp.a < 0;
412  });
413 
414  // initial lower bound on x
415  Scalar a = std::transform_reduce(
416  std::ranges::begin(hps_x_lower),
417  std::ranges::end(hps_x_lower),
418  -inf,
419  [](const Scalar a, const Scalar b) { return std::max(a, b); },
420  [](const HalfPlane & hp) { return hp.c / hp.a; });
421 
422  // halfplanes that define an upper bound on x (independent of y)
423  auto hps_x_upper = hps | std::views::filter([](const auto & hp) {
424  return hp.active && std::abs(hp.b) < eps && hp.a > 0;
425  });
426 
427  // initial upper bound on x
428  Scalar b = std::transform_reduce(
429  std::ranges::begin(hps_x_upper),
430  std::ranges::end(hps_x_upper),
431  inf,
432  [](const Scalar a, const Scalar b) { return std::min(a, b); },
433  [](const HalfPlane & hp) { return hp.c / hp.a; });
434 
435  // we remove at least one halfplane per iterations, so need at most N iterations
436  for (auto iter = hps.size(); iter > 0; --iter) {
437  const auto x = find_candidate(hps, a, b);
438 
439  // remove hps that were marked as not active
440  hps.erase(
441  std::remove_if(hps.begin(), hps.end(), [](const auto & hp) { return !hp.active; }),
442  hps.end());
443 
444  if (!x.has_value()) { break; }
445 
446  switch (check(hps, *x)) {
447  case 0:
448  return {*x, std::get<0>(gfun(hps, *x)), Status::Optimal};
449  break;
450  case 1:
451  b = *x;
452  break;
453  case 2:
454  a = *x;
455  break;
456  case 3:
457  return {0, inf, Status::PrimaryInfeasible};
458  break;
459  }
460  }
461 
462  // no intersection points, only need to consider boundaries
463  const auto [ga, sga, Sga] = gfun(hps, a);
464  const auto [ha, sha, Sha] = hfun(hps, a);
465 
466  const auto [gb, sgb, Sgb] = gfun(hps, b);
467  const auto [hb, shb, Shb] = hfun(hps, b);
468 
469  if (ga == ha && gb == hb && std::abs(ga) == inf && std::abs(gb) == inf) {
470  // special case where bounds are equal and \pm inf
471  if (gfun(hps, 0) <= hfun(hps, 0)) {
472  return {0., 0., Status::DualInfeasible};
473  } else {
474  return {0., 0., Status::PrimaryInfeasible};
475  }
476  }
477 
478  if (ga > ha && gb > hb) {
479  return {0, 0, Status::PrimaryInfeasible};
480  } else if ((ga <= ha && gb > hb) || ga < gb) {
481  return {a, ga, ga == -inf ? Status::DualInfeasible : Status::Optimal};
482  } else {
483  return {b, gb, gb == -inf ? Status::DualInfeasible : Status::Optimal};
484  }
485  }
