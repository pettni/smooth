89template<LieGroup G>
90inline Eigen::SparseMatrix<Scalar<G>> d_exp_sparse_pattern = [] {
91 Eigen::SparseMatrix<Scalar<G>> ret(Dof<G>, Dof<G>);
92 if constexpr (IsCommutative<G>) {
93 // identity matrix
94 for (auto i = 0u; i < Dof<G>; ++i) { ret.insert(i, i) = Scalar<G>(1); }
95 } else if constexpr (requires { traits::lie_sparse<G>::d_exp_sparse_pattern; }) {
96 // use specialized method if one exists
97 ret = traits::lie_sparse<G>::d_exp_sparse_pattern;
98 } else {
99 // fall back on dense pattern
100 for (auto i = 0u; i < Dof<G>; ++i) {
101 for (auto j = 0u; j < Dof<G>; ++j) { ret.insert(i, j) = (i == j) ? Scalar<G>(1) : Scalar<G>(0); }
102 }
103 }
104 ret.makeCompressed();
105 return ret;
106}();
