25template<std::size_t K, typename Scalar>
26constexpr StaticMatrix<Scalar, 1, K + 1> monomial_derivative(Scalar u, std::size_t p = 0)
27{
28 StaticMatrix<Scalar, 1, K + 1> ret;
29
30 if (p > K) { return ret; }
31
32 for (auto i = 0u; i < p; ++i) { ret[0][i] = Scalar(0); }
33 Scalar P1 = 1;
34 std::size_t P2 = 1;
35 for (auto j = 2u; j <= p; ++j) { P2 *= j; }
36 ret[0][p] = P1 * static_cast<Scalar>(P2);
37 for (auto i = p + 1; i <= K; ++i) {
38 P1 *= u;
39 P2 *= i;
40 P2 /= i - p;
41 ret[0][i] = P1 * static_cast<Scalar>(P2);
42 }
43
44 return ret;
45}

73template<std::size_t K, typename Scalar = double>
74constexpr StaticMatrix<Scalar, K + 1, K + 1> bspline_basis()
75{
76 StaticMatrix<Scalar, K + 1, K + 1> ret;
77 if constexpr (K == 0) {
78 ret[0][0] = 1;
79 return ret;
80 } else {
81 const auto coeff_mat_km1 = bspline_basis<K - 1, Scalar>();
82 StaticMatrix<Scalar, K + 1, K> low, high;
83 StaticMatrix<Scalar, K, K + 1> left, right;
84
85 for (auto i = 0u; i != K; ++i) {
86 for (auto j = 0u; j != K; ++j) {
87 low[i][j] = coeff_mat_km1[i][j];
88 high[i + 1][j] = coeff_mat_km1[i][j];
89 }
90 }
91
92 for (auto k = 0u; k != K; ++k) {
93 left[k][k + 1] = static_cast<Scalar>(K - (k + 1)) / static_cast<Scalar>(K);
94 left[k][k] = Scalar(1) - left[k][k + 1];
95
96 right[k][k + 1] = Scalar(1) / static_cast<Scalar>(K);
97 right[k][k] = -right[k][k + 1];
98 }
99
100 return low * left + high * right;
101 }
102}

108template<std::size_t K, typename Scalar = double>
109constexpr StaticMatrix<Scalar, K + 1, K + 1> bernstein_basis()
110{
111 StaticMatrix<Scalar, K + 1, K + 1> ret;
112 if constexpr (K == 0) {
113 ret[0][0] = 1;
114 return ret;
115 } else {
116 constexpr auto coeff_mat_km1 = bernstein_basis<K - 1, Scalar>();
117 StaticMatrix<Scalar, K + 1, K> low, high;
118 StaticMatrix<Scalar, K, K + 1> left, right;
119
120 for (auto i = 0u; i != K; ++i) {
121 for (auto j = 0u; j != K; ++j) {
122 low[i][j] = coeff_mat_km1[i][j];
123 high[i + 1][j] = coeff_mat_km1[i][j];
124 }
125 }
126
127 for (auto k = 0u; k != K; ++k) {
128 left[k][k] = Scalar(1);
129
130 right[k][k] = Scalar(-1);
131 right[k][k + 1] = Scalar(1);
132 }
133
134 return low * left + high * right;
135 }

142template<std::size_t K, typename Scalar = double>
143constexpr StaticMatrix<Scalar, K + 1, K + 1> hermite_basis()
144{
145 StaticMatrix<Scalar, K + 1, K + 1> ret;
146 ret[0][0] = 1;
147
148 if constexpr (K > 0) { ret[1][1] = 2; }
149
150 if constexpr (K > 1) {
151 for (auto k = 2u; k < K + 1; ++k) {
152 for (auto i = 0u; i < k; ++i) { ret[i + 1][k] += 2 * ret[i][k - 1]; }
153 for (auto i = 0u; i + 1 < k; ++i) { ret[i][k] -= 2 * (k - 1) * ret[i][k - 2]; }
154 }
155 }
156 return ret;
157}

163template<std::size_t K, typename Scalar = double>
164constexpr StaticMatrix<Scalar, K + 1, K + 1> laguerre_basis()
165{
166 StaticMatrix<Scalar, K + 1, K + 1> ret;
167 ret[0][0] = 1;
168
169 if constexpr (K > 0) {
170 ret[0][1] = 1.;
171 ret[1][1] = -1;
172 }
173
174 if constexpr (K > 1) {
175 for (auto k = 2u; k < K + 1; ++k) {
176 for (auto i = 0u; i < k; ++i) {
177 ret[i][k] += (2 * k - 1) * ret[i][k - 1] / k;
178 ret[i + 1][k] -= ret[i][k - 1] / k;
179 }
180 for (auto i = 0u; i + 1 < k; ++i) { ret[i][k] -= static_cast<double>(k - 1) * ret[i][k - 2] / k; }
181 }
182 }
183 return ret;
184}

197template<std::size_t K, typename Scalar = double>
198constexpr StaticMatrix<Scalar, K + 1, K + 1> jacobi_basis(double alpha, double beta)
199{
200 StaticMatrix<Scalar, K + 1, K + 1> ret;
201 ret[0][0] = 1;
202
203 if constexpr (K > 0) {
204 ret[0][1] = alpha + 1. - (alpha + beta + 2) / 2;
205 ret[1][1] = (alpha + beta + 2.) / 2;
206 }
207
208 if constexpr (K > 1) {
209 for (auto k = 2u; k < K + 1; ++k) {
210 const auto frac = 1. / ((2 * k) * (k + alpha + beta) * (2 * k + alpha + beta - 2));
211 const auto c1 = (2 * k + alpha + beta - 1) * (alpha * alpha - beta * beta);
212 const auto c2 = (2 * k + alpha + beta - 1) * (2 * k + alpha + beta) * (2 * k + alpha + beta - 2);
213 const auto c3 = 2 * (k + alpha - 1) * (k + beta - 1) * (2 * k + alpha + beta);
214 for (auto i = 0u; i < k; ++i) {
215 ret[i][k] += c1 * ret[i][k - 1] * frac;
216 ret[i + 1][k] += c2 * ret[i][k - 1] * frac;
217 }
218 for (auto i = 0u; i + 1 < k; ++i) { ret[i][k] -= c3 * ret[i][k - 2] * frac; }
219 }
220 }
221
222 return ret;
223}

262template<PolynomialBasis Basis, std::size_t K, typename Scalar = double>
263constexpr StaticMatrix<Scalar, K + 1, K + 1> polynomial_basis()
264{
265 if constexpr (Basis == PolynomialBasis::Monomial) {
266 StaticMatrix<Scalar, K + 1, K + 1> ret;
267 for (auto k = 0u; k <= K; ++k) { ret[k][k] = Scalar(1); }
268 return ret;
269 }
270 if constexpr (Basis == PolynomialBasis::Bernstein) { return detail::bernstein_basis<K, Scalar>(); }
271 if constexpr (Basis == PolynomialBasis::Laguerre) { return detail::laguerre_basis<K, Scalar>(); }
272 if constexpr (Basis == PolynomialBasis::Hermite) { return detail::hermite_basis<K, Scalar>(); }
273 if constexpr (Basis == PolynomialBasis::Legendre) { return detail::jacobi_basis<K, Scalar>(0, 0); }
274 if constexpr (Basis == PolynomialBasis::Chebyshev1st) {
275 auto ret = detail::jacobi_basis<K, Scalar>(-0.5, -0.5);
276 const auto fac = monomial_derivative<K>(1.) * ret;
277 for (auto k = 0u; k < K + 1; ++k) {
278 for (auto r = 0u; r < K + 1; ++r) { ret[r][k] /= fac[0][k]; }
279 }
280 return ret;
281 }
282 if constexpr (Basis == PolynomialBasis::Chebyshev2nd) {
283 auto ret = detail::jacobi_basis<K, Scalar>(0.5, 0.5);
284 const auto fac = monomial_derivative<K>(1.) * ret;
285 for (auto k = 0u; k < K + 1; ++k) {
286 for (auto r = 0u; r < K + 1; ++r) { ret[r][k] *= static_cast<Scalar>(k + 1) / fac[0][k]; }
287 }
288 return ret;
289 }
290 if constexpr (Basis == PolynomialBasis::Bspline) { return detail::bspline_basis<K, Scalar>(); }
291}

311template<std::size_t K, std::ranges::random_access_range R, typename Scalar = std::ranges::range_value_t<R>>
312constexpr StaticMatrix<Scalar, K + 1, K + 1> lagrange_basis(const R & ts)
313{
314 StaticMatrix<Scalar, K + 1, K + 1> ret;
315
316 for (auto row = 0u; row < K + 1; ++row) {
317 ret[row][0] = 1;
318 for (auto col = 0u; col < K + 1; ++col) {
319 if (col != row) {
320 const auto row_copy = ret[row];
321 ret[row].fill(0);
322 for (auto i = 0u; i <= col - (col > row ? 1 : 0); ++i) {
323 ret[row][i + 1] += row_copy[i] / (*(ts.begin() + row) - *(ts.begin() + col));
324 ret[row][i] -= *(ts.begin() + col) * row_copy[i] / (*(ts.begin() + row) - *(ts.begin() + col));
325 }
326 }
327 }
328 }
329
330 return ret.transpose();
331}

344template<std::size_t K, std::size_t N, typename Scalar, std::ranges::sized_range R>
345constexpr StaticMatrix<Scalar, K + 1, N>
346polynomial_basis_derivatives(const StaticMatrix<Scalar, K + 1, K + 1> & B, const R & ts)
347{
348 assert(std::ranges::size(ts) == N);
349
350 StaticMatrix<Scalar, K + 1, N> ret;
351
352 for (auto j = 0u; const auto t : ts) {
353 const auto U = monomial_derivative<K>(t, 1);
354 const auto U_B = U * B;
355 for (auto i = 0u; i < K + 1; ++i) { ret[i][j] = U_B[0][i]; }
356 ++j;
357 }
358
359 return ret;
360}

398template<std::size_t K, std::size_t P, typename Scalar = double>
399constexpr StaticMatrix<Scalar, K + 1, K + 1> monomial_integral()
400{
401 StaticMatrix<Scalar, K + 1, K + 1> ret;
402 for (auto i = 0u; i <= K; ++i) {
403 for (auto j = i; j <= K; ++j) {
404 if (i >= P && j >= P) {
405 std::size_t c = 1;
406 for (auto _i = i - P + 1; _i <= i; ++_i) { c *= _i; }
407 for (auto _j = j - P + 1; _j <= j; ++_j) { c *= _j; }
408 ret[i][j] = static_cast<Scalar>(c) / static_cast<Scalar>(i + j - 2 * P + 1);
409 } else {
410 ret[i][j] = 0;
411 }
412 ret[j][i] = ret[i][j];
413 }
414 }
415 return ret;
416}

426inline constexpr double integrate_absolute_polynomial(double t0, double t1, double A, double B, double C)
427{
428 // location of first zero (if any)
429 double mid1 = std::numeric_limits<double>::infinity();
430 // location of second zero (if any)
431 double mid2 = std::numeric_limits<double>::infinity();
432
433 if (std::abs(A) < 1e-9 && std::abs(B) > 1e-9) {
434 // linear non-constant function
435 mid1 = std::clamp(-C / B, t0, t1);
436 } else if (std::abs(A) > 1e-9) {
437 // quadratic function
438 const double res = B * B / (4 * A * A) - C / A;
439
440 if (res > 0) {
441 mid1 = -B / (2 * A) - std::sqrt(res);
442 mid2 = -B / (2 * A) + std::sqrt(res);
443 }
444 }
445
446 const auto integ = [&](double u) { return A * u * u * u / 3 + B * u * u / 2 + C * u; };
447
448 const double mid1cl = std::clamp(mid1, t0, t1);
449 const double mid2cl = std::clamp(mid2, t0, t1);
450
451 return std::abs(integ(t1) - integ(t0) + 2 * integ(mid1cl) - 2 * integ(mid2cl));
452}
