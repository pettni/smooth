33template<typename G>
34concept LieGroup =
35requires (Eigen::Index dof) {
36 // Underlying scalar type
37 typename traits::lie<G>::Scalar;
38 // Default representation
39 typename traits::lie<G>::PlainObject;
40 // Compile-time degrees of freedom (tangent space dimension). Can be dynamic (equal to -1)
41 {traits::lie<G>::Dof}->std::convertible_to<int>;
42 // Commutativity
43 {traits::lie<G>::IsCommutative}->std::convertible_to<bool>;
44 // Return the identity element (dof = Dof for static size)
45 {traits::lie<G>::Identity(dof)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
46 // Return a random element (dof = Dof for static size)
47 {traits::lie<G>::Random(dof)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
48} &&
49// GROUP INTERFACE
50requires(const G & g1, const G & g2, typename traits::lie<G>::Scalar eps) {
51 // Group adjoint
52 {traits::lie<G>::Ad(g1)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
53 // Group composition
54 {traits::lie<G>::composition(g1, g2)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
55 // Run-time degrees of freedom (tangent space dimension).
56 {traits::lie<G>::dof(g1)}->std::convertible_to<Eigen::Index>;
57 // Group inverse
58 {traits::lie<G>::inverse(g1)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
59 // Check if two elements are (approximately) equal
60 {traits::lie<G>::isApprox(g1, g2, eps)}->std::convertible_to<bool>;
61 // Group logarithm (maps from group to algebra)
62 {traits::lie<G>::log(g1)}->std::convertible_to<Eigen::Vector<typename traits::lie<G>::Scalar, traits::lie<G>::Dof>>;
63} &&
64// TANGENT INTERFACE
65requires(const Eigen::Vector<typename traits::lie<G>::Scalar, traits::lie<G>::Dof> & a) {
66 // Algebra adjoint
67 {traits::lie<G>::ad(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
68 // Algebra exponential (maps from algebra to group)
69 {traits::lie<G>::exp(a)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
70 // Right derivative of the exponential map
71 {traits::lie<G>::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
72 // Right derivative of the exponential map inverse
73 {traits::lie<G>::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
74 // Second right derivative of the exponential map
75 {traits::lie<G>::d2r_exp(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, (traits::lie<G>::Dof > 0 ? traits::lie<G>::Dof * traits::lie<G>::Dof : -1)>>;
76 // Second right derivative of the exponential map inverse
77 {traits::lie<G>::d2r_expinv(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, (traits::lie<G>::Dof > 0 ? traits::lie<G>::Dof * traits::lie<G>::Dof : -1)>>;
78} && (
79 // Cast to different scalar type
80 !std::is_convertible_v<typename traits::lie<G>::Scalar, double> ||
81 requires (const G & g) {
82 {traits::lie<G>::template cast<double>(g)};
83 }
84) && (
85 !std::is_convertible_v<typename traits::lie<G>::Scalar, float> ||
86 requires (const G & g) {
87 {traits::lie<G>::template cast<double>(g)};
88 }
89) &&
90// PlainObject must be default-constructible
91std::is_default_constructible_v<typename traits::lie<G>::PlainObject> &&
92std::is_copy_constructible_v<typename traits::lie<G>::PlainObject> &&
93// PlainObject must be assignable from G
94std::is_assignable_v<typename traits::lie<G>::PlainObject &, G>;
