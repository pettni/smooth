194  G eval(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
195  {
197  // TODO binary search
198  std::size_t istar = 0;
199  while (istar + 2 < knots_.size() && knots_[istar + 1] <= t) { ++istar; }
200 
201  double T = knots_[istar + 1] - knots_[istar];
202 
203  const double u = (t - knots_[istar]) / T;
204 
205  G g = segments_[istar].eval(u, vel, acc);
206 
207  if (vel.has_value()) { vel.value() /= T; }
208  if (acc.has_value()) { acc.value() /= (T * T); }
209 
210  return g;
211  }

230 template<std::ranges::range Rt, std::ranges::range Rg>
231 PiecewiseBezier<1, std::ranges::range_value_t<Rg>> fit_linear_bezier(const Rt & tt, const Rg & gg)
232 {
233  if (std::ranges::size(tt) < 2 || std::ranges::size(gg) < 2) {
234  throw std::runtime_error("Not enough points");
235  }
236  using G = std::ranges::range_value_t<Rg>;
237 
238  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
239 
240  std::vector<Bezier<1, G>> segments(N);
241 
242  auto it_g = std::ranges::begin(gg);
243  auto it_t = std::ranges::begin(tt);
244 
245  for (auto i = 0u; i != N; ++i, ++it_t, ++it_g) {
246  segments[i] = Bezier<1, G>(*it_g, std::array<typename G::Tangent, 1>{(*(it_g + 1) - *it_g)});
247  }
248 
249  auto take_view = tt | std::views::take(N + 1);
250  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
251 
252  return PiecewiseBezier<1, G>(std::move(knots), std::move(segments));
253 }

268 template<std::ranges::range Rt, std::ranges::range Rg>
269 PiecewiseBezier<2, std::ranges::range_value_t<Rg>> fit_quadratic_bezier(
270  const Rt & tt, const Rg & gg)
271 {
272  if (std::ranges::size(tt) < 2 || std::ranges::size(gg) < 2) {
273  throw std::runtime_error("Not enough points");
274  }
275  using G = std::ranges::range_value_t<Rg>;
276  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
277 
278  std::vector<Bezier<2, G>> segments(N);
279 
280  auto it_g = std::ranges::begin(gg);
281  auto it_t = std::ranges::begin(tt);
282 
283  typename G::Tangent v0 = (*(it_g + 1) - *it_g) / (*(it_t + 1) - *it_t);
284 
285  for (auto i = 0u; i != N; ++i, ++it_t, ++it_g) {
286  const double dt = *(it_t + 1) - *it_t;
287 
288  // scaled velocity
289  const typename G::Tangent va = v0 * dt;
290 
291  // create segment
292  const typename G::Tangent v1 = va / 2;
293  const typename G::Tangent v2 = *(it_g + 1) - (*it_g * G::exp(va / 2));
294 
295  segments[i] = Bezier<2, G>(*it_g, std::array<typename G::Tangent, 2>{v1, v2});
296 
297  // unscaled end velocity for interval
298  v0 = v2 * 2 / dt;
299  }
300 
301  auto take_view = tt | std::views::take(N + 1);
302  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
303 
304  return PiecewiseBezier<2, G>(std::move(knots), std::move(segments));
305 }

317 template<std::ranges::range Rt, std::ranges::range Rg>
318 PiecewiseBezier<3, std::ranges::range_value_t<Rg>> fit_cubic_bezier(const Rt & tt, const Rg & gg)
319 {
320  if (std::ranges::size(tt) < 2 || std::ranges::size(gg) < 2) {
321  throw std::runtime_error("Not enough points");
322  }
323  using G = std::ranges::range_value_t<Rg>;
324 
325  // number of intervals
326  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
327 
328  std::size_t NumVars = G::Dof * 3 * N;
329 
330  Eigen::SparseMatrix<typename G::Scalar> lhs;
331  lhs.resize(NumVars, NumVars);
332  Eigen::Matrix<int, -1, 1> nnz = Eigen::Matrix<int, -1, 1>::Constant(NumVars, 3);
333  nnz.head(G::Dof).setConstant(2);
334  nnz.tail(G::Dof).setConstant(2);
335  lhs.reserve(nnz);
336 
337  Eigen::Matrix<typename G::Scalar, -1, 1> rhs(NumVars);
338  rhs.setZero();
339 
340  // variable layout:
341  //
342  // [ v_{1, 0}; v_{2, 0}; v_{3, 0}; v_{1, 1}; v_{2, 1}; v_{3, 1}; ...]
343  //
344  // where v_ji is a Dof-length vector
345 
346  const auto idx = [&](int j, int i) { return 3 * G::Dof * i + G::Dof * (j - 1); };
347 
348  std::size_t row_counter = 0;
349 
351 
352  // zero second derivative at start:
353  // v_{1, 0} = v_{2, 0}
354  const std::size_t v10_start = idx(1, 0);
355  const std::size_t v20_start = idx(2, 0);
356  for (auto n = 0u; n != G::Dof; ++n) {
357  lhs.insert(row_counter + n, v10_start + n) = 1;
358  lhs.insert(row_counter + n, v20_start + n) = -1;
359  }
360  row_counter += G::Dof;
361 
363 
364  auto it_t = std::ranges::begin(tt);
365  auto it_g = std::ranges::begin(gg);
366 
367  for (auto i = 0u; i + 1 < N; ++i, ++it_t, ++it_g) {
368  const std::size_t v1i_start = idx(1, i);
369  const std::size_t v2i_start = idx(2, i);
370  const std::size_t v3i_start = idx(3, i);
371 
372  const std::size_t v1ip_start = idx(1, i + 1);
373  const std::size_t v2ip_start = idx(2, i + 1);
374 
375  // segment lengths
376  const typename G::Scalar Ti = *(it_t + 1) - *it_t;
377  const typename G::Scalar Tip = *(it_t + 2) - *(it_t + 1);
378 
379  // pass through control points
380  // v_{1, i} + v_{2, i} + v_{3, i} = x_{i+1} - x_i
381  for (auto n = 0u; n != G::Dof; ++n) {
382  lhs.insert(row_counter + n, v1i_start + n) = 1;
383  lhs.insert(row_counter + n, v2i_start + n) = 1;
384  lhs.insert(row_counter + n, v3i_start + n) = 1;
385  }
386  rhs.segment(row_counter, G::Dof) = *(it_g + 1) - *(it_g);
387  row_counter += G::Dof;
388 
389  // velocity continuity
390  // v_{3, i} = v_{1, i+1}
391  for (auto n = 0u; n != G::Dof; ++n) {
392  lhs.insert(row_counter + n, v3i_start + n) = 1 * Tip;
393  lhs.insert(row_counter + n, v1ip_start + n) = -1 * Ti;
394  }
395  row_counter += G::Dof;
396 
397  // acceleration continuity (approximate for Lie groups)
398  // v_{2, i} - v_{3, i} = v_{2, i+1} - v_{1, i+1}
399  for (auto n = 0u; n != G::Dof; ++n) {
400  lhs.insert(row_counter + n, v2i_start + n) = 1 * (Tip * Tip);
401  lhs.insert(row_counter + n, v3i_start + n) = -1 * (Tip * Tip);
402  lhs.insert(row_counter + n, v1ip_start + n) = -1 * (Ti * Ti);
403  lhs.insert(row_counter + n, v2ip_start + n) = 1 * (Ti * Ti);
404  }
405  row_counter += G::Dof;
406  }
407 
409 
410  const std::size_t v1_nm_start = idx(1, N - 1);
411  const std::size_t v2_nm_start = idx(2, N - 1);
412  const std::size_t v3_nm_start = idx(3, N - 1);
413 
414  // end at last control point
415  // v_{1, n-1} + v_{2, n-1} v_{3, n-1} = x_{n} - x_{n-1}
416  for (auto n = 0u; n != G::Dof; ++n) {
417  lhs.insert(row_counter + n, v1_nm_start + n) = 1;
418  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
419  lhs.insert(row_counter + n, v3_nm_start + n) = 1;
420  }
421  rhs.segment(row_counter, G::Dof) = *(it_g + 1) - *it_g;
422  row_counter += G::Dof;
423 
424  // zero second derivative at end:
425  // v_{2, n-1} = v_{3, n-1}
426  for (auto n = 0u; n != G::Dof; ++n) {
427  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
428  lhs.insert(row_counter + n, v3_nm_start + n) = -1;
429  }
430 
432 
433  lhs.makeCompressed();
434 
436 
437  Eigen::SparseLU<decltype(lhs), Eigen::COLAMDOrdering<int>> solver(lhs);
438  Eigen::VectorXd result = solver.solve(rhs);
439 
441 
442  std::vector<Bezier<3, G>> segments;
443  segments.reserve(N);
444 
445  it_g = std::ranges::begin(gg);
446 
447  for (auto i = 0u; i != N; ++i, ++it_g) {
448  const std::size_t v1i_start = idx(1, i);
449  const std::size_t v3i_start = idx(3, i);
450 
451  typename G::Tangent v1 = result.template segment<G::Dof>(v1i_start);
452  typename G::Tangent v3 = result.template segment<G::Dof>(v3i_start);
453  // re-compute v2 to compensate for linearization
454  // this ensures points are interpolated, but the cost is
455  // potential non-continuity of the second derivative
456  typename G::Tangent v2 = *(it_g + 1) * G::exp(-v3) - *(it_g)*G::exp(v1);
457 
458  segments.emplace_back(
459  *it_g, std::array<typename G::Tangent, 3>{std::move(v1), std::move(v2), std::move(v3)});
460  }
461 
462  auto take_view = tt | std::views::take(N + 1);
463  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
464 
465  return PiecewiseBezier<3, G>(std::move(knots), std::move(segments));
466 }
